def two_sum_sorted(arr, target):
    l, r = 0, len(arr) - 1
    while l < r:
        s = arr[l] + arr[r]
        if s == target:
            return [l, r]
        elif s < target:
            l += 1
        else:
            r -= 1

def longest_subarray_sum(arr, k):
    l = s = ans = 0
    for r in range(len(arr)):
        s += arr[r]
        while s > k:
            s -= arr[l]
            l += 1
        ans = max(ans, r - l + 1)
    return ans
